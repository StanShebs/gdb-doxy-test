00023 init_target_desc (struct target_desc *tdesc)
00024 {
00025   int offset, i;
00026 
00027   offset = 0;
00028   for (i = 0; i < tdesc->num_registers; i++)
00029     {
00030       tdesc->reg_defs[i].offset = offset;
00031       offset += tdesc->reg_defs[i].size;
00032     }
00033 
00034   tdesc->registers_size = offset / 8;
00035 
00036   /* Make sure PBUFSIZ is large enough to hold a full register
00037      packet.  */
00038   if (2 * tdesc->registers_size + 32 > PBUFSIZ)
00039     fatal ("Register packet size exceeds PBUFSIZ.");
00040 }
