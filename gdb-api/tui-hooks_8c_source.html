00093       clearerr (stdin);         /* in case of C-d */
00094       if (answer == EOF)        /* C-d */
00095         {
00096           retval = 1;
00097           break;
00098         }
00099       /* Eat rest of input line, to EOF or newline.  */
00100       if (answer != '\n')
00101         do
00102           {
00103             ans2 = tui_getc (stdin);
00104             clearerr (stdin);
00105           }
00106         while (ans2 != EOF && ans2 != '\n' && ans2 != '\r');
00107 
00108       if (answer >= 'a')
00109         answer -= 040;
00110       if (answer == 'Y')
00111         {
00112           retval = 1;
00113           break;
00114         }
00115       if (answer == 'N')
00116         {
00117           retval = 0;
00118           break;
00119         }
