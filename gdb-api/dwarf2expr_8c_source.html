00636                   const gdb_byte *op_ptr, const gdb_byte *op_end)
00637 {
00638   enum bfd_endian byte_order = gdbarch_byte_order (ctx->gdbarch);
00639   /* Old-style "untyped" DWARF values need special treatment in a
00640      couple of places, specifically DW_OP_mod and DW_OP_shr.  We need
00641      a special type for these values so we can distinguish them from
