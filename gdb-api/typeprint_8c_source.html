00113 recursively_update_typedef_hash (struct typedef_hash_table *table,
00114                                  struct type *t)
00115 {

00117

00121   for (i = 0; i < TYPE_TYPEDEF_FIELD_COUNT (t); ++i)

00517    currently use it, and it wasn't clear if it really belonged somewhere
