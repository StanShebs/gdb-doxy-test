00067 /* Return whether THIS_FRAME corresponds to a Solaris sigtramp
00068    routine.  */
00069 
00070 static int
00071 amd64_sol2_sigtramp_p (struct frame_info *this_frame)
00072 {
00073   CORE_ADDR pc = get_frame_pc (this_frame);
00074   const char *name;
00075 
00076   find_pc_partial_function (pc, &name, NULL, NULL);
00077   return (name && (strcmp ("sigacthandler", name) == 0
00078                    || strcmp (name, "ucbsigvechandler") == 0));
00079 }

00081 /* Solaris doesn't have a 'struct sigcontext', but it does have a
00082    'mcontext_t' that contains the saved set of machine registers.  */
00083 
00084 static CORE_ADDR
00085 amd64_sol2_mcontext_addr (struct frame_info *this_frame)
00086 {
00087   CORE_ADDR sp, ucontext_addr;
00088 
00089   sp = get_frame_register_unsigned (this_frame, AMD64_RSP_REGNUM);
00090   ucontext_addr = get_frame_memory_unsigned (this_frame, sp + 8, 8);
00091 
00092   return ucontext_addr + 72;
00093 }
