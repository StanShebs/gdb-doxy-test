00117 /* Type of floating-point code in use by inferior.  There are really 3 models
00118    that are traditionally supported (plus the endianness issue), but gcc can
00119    only generate 2 of those.  The third is APCS_FLOAT, where arguments to
00120    functions are passed in floating-point registers.  
00121 
00122    In addition to the traditional models, VFP adds two more. 
00123 
00124    If you update this enum, don't forget to update fp_model_strings in 
00125    arm-tdep.c.  */

00128 {

00130   ARM_FLOAT_SOFT_FPA,   /* Traditional soft-float (mixed-endian on LE ARM).  */
00131   ARM_FLOAT_FPA,        /* FPA co-processor.  GCC calling convention.  */
00132   ARM_FLOAT_SOFT_VFP,   /* Soft-float with pure-endian doubles.  */
00133   ARM_FLOAT_VFP,        /* Full VFP calling convention.  */
00134   ARM_FLOAT_LAST        /* Keep at end.  */
