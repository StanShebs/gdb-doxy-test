00171 /* We can't really make windows visible, or invisible.  So we have to
00172    delete the entire window when making it visible, and create it
00173    again when making it visible.  */
00174 static void
00175 make_visible (struct tui_gen_win_info *win_info, int visible)
00176 {
00177   /* Don't tear down/recreate command window.  */
00178   if (win_info->type == CMD_WIN)
00179     return;
00180 
00181   if (visible)
00182     {
00183       if (!win_info->is_visible)
00184         {
00185           tui_make_window (win_info,
00186                            (win_info->type != CMD_WIN
00187                             && !tui_win_is_auxillary (win_info->type)));
00188           win_info->is_visible = TRUE;
00189         }
00190     }
00191   else if (!visible
00192            && win_info->is_visible
00193            && win_info->handle != (WINDOW *) NULL)
00194     {
00195       win_info->is_visible = FALSE;
00196       tui_delete_win (win_info->handle);
00197       win_info->handle = (WINDOW *) NULL;
00198     }
00199 
00200   return;
00201 }
