00039 static void
00040 arm_netbsd_init_abi_common (struct gdbarch_info info,
00041                             struct gdbarch *gdbarch)
00042 {
00043   struct gdbarch_tdep *tdep = gdbarch_tdep (gdbarch);
00044 
00045   tdep->lowest_pc = 0x8000;
00046   switch (info.byte_order)
00047     {
00048     case BFD_ENDIAN_LITTLE:
00049       tdep->arm_breakpoint = arm_nbsd_arm_le_breakpoint;
00050       tdep->thumb_breakpoint = arm_nbsd_thumb_le_breakpoint;
00051       tdep->arm_breakpoint_size = sizeof (arm_nbsd_arm_le_breakpoint);
00052       tdep->thumb_breakpoint_size = sizeof (arm_nbsd_thumb_le_breakpoint);
00053       break;
00054 
00055     case BFD_ENDIAN_BIG:
00056       tdep->arm_breakpoint = arm_nbsd_arm_be_breakpoint;
00057       tdep->thumb_breakpoint = arm_nbsd_thumb_be_breakpoint;
00058       tdep->arm_breakpoint_size = sizeof (arm_nbsd_arm_be_breakpoint);
00059       tdep->thumb_breakpoint_size = sizeof (arm_nbsd_thumb_be_breakpoint);
00060       break;
00061 
00062     default:
00063       internal_error (__FILE__, __LINE__,
00064                       _("arm_gdbarch_init: bad byte order for float format"));
00065     }
00066 
00067   tdep->jb_pc = ARM_NBSD_JB_PC;
00068   tdep->jb_elt_size = ARM_NBSD_JB_ELEMENT_SIZE;
00069 
00070   /* Single stepping.  */
00071   set_gdbarch_software_single_step (gdbarch, arm_software_single_step);
