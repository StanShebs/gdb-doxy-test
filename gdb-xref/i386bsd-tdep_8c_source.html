00036 static CORE_ADDR
00037 i386bsd_sigcontext_addr (struct frame_info *this_frame)
00038 {
00039   struct gdbarch *gdbarch = get_frame_arch (this_frame);
00040   enum bfd_endian byte_order = gdbarch_byte_order (gdbarch);
00041   gdb_byte buf[4];
00042   CORE_ADDR sp;
00043 
00044   get_frame_register (this_frame, I386_ESP_REGNUM, buf);
00045   sp = extract_unsigned_integer (buf, 4, byte_order);
00046 
00047   return read_memory_unsigned_integer (sp + 8, 4, byte_order);
