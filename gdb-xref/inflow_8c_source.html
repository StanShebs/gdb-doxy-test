00691 #if !defined(__GO32__) && !defined(_WIN32)

00702       ioctl (tty, TIOCNOTTY, 0);
00703       close (tty);
00704       signal (SIGTTOU, osigttou);
00705     }
00706 #endif

00709   tty = open (inferior_thisrun_terminal, O_RDWR | O_NOCTTY);
00710   check_syscall (inferior_thisrun_terminal, tty);

00712   /* Avoid use of dup2; doesn't exist on all systems.  */
00713   if (tty != 0)
00714     {
00715       close (0);
00716       check_syscall ("dup'ing tty into fd 0", dup (tty));
00717     }
00718   if (tty != 1)
00719     {
00720       close (1);
00721       check_syscall ("dup'ing tty into fd 1", dup (tty));
00722     }
00723   if (tty != 2)
00724     {
00725       close (2);
00726       check_syscall ("dup'ing tty into fd 2", dup (tty));
00727     }
00728 
00729 #ifdef TIOCSCTTY
00730   /* Make tty our new controlling terminal.  */
